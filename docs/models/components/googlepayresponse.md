# GooglePayResponse

Describes a Google Pay token on a Moov account.

## Example Usage

```typescript
import { GooglePayResponse } from "@moovio/sdk/models/components";

let value: GooglePayResponse = {
  tokenID: "dcd471ec-7d5c-4016-865c-1e78c733bea2",
  brand: "Visa",
  cardType: "credit",
  cardDisplayName: "7419",
  fingerprint:
    "9948962d92a1ce40c9f918cd9ece3a22bde62fb325a2f1fe2e833969de672ba3",
  expiration: {
    month: "01",
    year: "21",
  },
  dynamicLastFour: "7419",
  issuerCountry: "US",
  authMethod: "PAN_ONLY",
};
```
