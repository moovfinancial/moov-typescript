# CardMetadata

Metadata describing a card.

## Example Usage

```typescript
import { CardMetadata } from "@moovio/sdk/models/components";

let value: CardMetadata = {
  bin: "411111",
  brand: "Visa",
  cardCategory: "CLASSIC",
  cardType: "credit",
  commercial: false,
  regulated: false,
  issuer: "GRINGOTTS BANK",
  issuerCountry: "US",
  issuerPhone: "8185551212",
  issuerURL: "HTTPS://WWW.EXAMPLE.COM/",
  domesticPullFromCard: "supported",
  domesticPushToCard: "standard",
};
```
