# FullIssuedCard

The full details of an issued card, including PAN and CVV.

## Example Usage

```typescript
import { FullIssuedCard } from "moov-sdk/models/components";

let value: FullIssuedCard = {
  issuedCardID: "4dcb0047-e649-4ebd-b228-bf558a9b6d25",
  brand: "Visa",
  lastFourCardNumber: "<value>",
  expiration: {
    month: "01",
    year: "21",
  },
  authorizedUser: {
    firstName: "Etha",
    lastName: "Green",
  },
  fundingWalletID: "<id>",
  state: "active",
  formFactor: "virtual",
  controls: {
    velocityLimits: [
      {
        amount: 10000,
        interval: "per-transaction",
      },
    ],
  },
  createdOn: new Date("2023-11-26T14:28:09.704Z"),
  pan: "<value>",
  cvv: "842",
};
```
