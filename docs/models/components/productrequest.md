# ProductRequest

Request to create or update a product.

## Example Usage

```typescript
import { ProductRequest } from "@moovio/sdk/models/components";

let value: ProductRequest = {
  title: "<value>",
  basePrice: {
    currency: "USD",
    valueDecimal: "12.987654321",
  },
  optionGroups: [
    {
      name: "<value>",
      minSelect: 903729,
      maxSelect: 102098,
      options: [
        {
          name: "<value>",
          priceModifier: "-14.89",
        },
      ],
    },
  ],
};
```
